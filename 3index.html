body { font-family: 'Segoe UI', sans-serif; background-color: #f4f4f4; display: flex; justify-content: center; padding: 20px; } .quiz-container { background: white; padding: 30px; border-radius: 10px; box-shadow: 0 0 10px rgba(0,0,0,0.1); width: 400px; } h1 { text-align: center; color: #333; } .question { font-weight: bold; margin-bottom: 10px; } .answers { margin-bottom: 20px; } button { width: 100%; padding: 10px; background: #28a745; color: white; border: none; border-radius: 5px; cursor: pointer; } button:hover { background: #218838; } #results { margin-top: 20px; font-weight: bold; text-align: center; font-size: 1.2em; } const quizData = [ { question: "1. የዓባይ ወንዝ መነሻ የት ነው?", a: "ጣና ሐይቅ", b: "ግሸ ዓባይ", c: "ሱዳን", correct: "b" }, { question: "2. የኢትዮጵያ ቀለሞች ቅደም ተከተል?", a: "ቢጫ፣ ቀይ፣ አረንጓዴ", b: "ቀይ፣ ቢጫ፣ አረንጓዴ", c: "አረንጓዴ፣ ቢጫ፣ ቀይ", correct: "c" } ]; const quizContainer = document.getElementById('quiz'); const resultsContainer = document.getElementById('results'); const submitButton = document.getElementById('submit'); // ጥያቄዎቹን በገጹ ላይ ማሳያ function buildQuiz() { let output = ""; quizData.forEach((data, index) => { output += `
${data.question}
${data.a}
${data.b}
${data.c}
`; }); quizContainer.innerHTML = output; } // ውጤት ማስያ function showResults() { let score = 0; quizData.forEach((data, index) => { const selector = `input[name="q${index}"]:checked`; const userAnswer = (document.querySelector(selector) || {}).value; if (userAnswer === data.correct) { score++; } }); resultsContainer.innerHTML = `ውጤትህ፦ ${score} ከ ${quizData.length}`; } buildQuiz(); submitButton.addEventListener('click', showResults);
